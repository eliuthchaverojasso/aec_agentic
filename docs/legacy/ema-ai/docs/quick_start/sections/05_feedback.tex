\section{Submit Feedback}
Users must submit one Google Form response for each requirement that needs review.

\subsection*{Feedback form}
\begin{center}
\fcolorbox{EMA Blue}{EMA SoftBlue}{\parbox{0.95\textwidth}{\centering\link{https://forms.gle/6JCgsHnbua8pn7mu9}}}
\end{center}

\subsection*{Required fields}
\begin{itemize}
  \item Project Name
  \item Reviewer Name
  \item Requirement ID
  \item Current Result
  \item Expected Result
  \item Supporting Reason
  \item Expected Evidence
\end{itemize}

\subsection*{Recommended but optional}
\begin{itemize}
  \item Screenshot
  \item Revit Element ID
  \item Drawing sheet
  \item Specification section
\end{itemize}
